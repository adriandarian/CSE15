    \tab $x \in X, y \in Y, z \in Z$ \\
    \tab $|X| = |Y|$ and $|Y| = |Z|$ \\
    \tab $\therefore f: X \rightarrow Y$ and $g: Y \rightarrow Z$ are bijections \\
    \tab $y \in Y$ such $g(y) = z$ and $x \in X$ such $f(x) = y$ \\

    \tab $g \circ f: X \rightarrow Z$ is a bijection \\
